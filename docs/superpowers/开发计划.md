  CREATE TABLE IF NOT EXISTS meeting_minutes (

      title VARCHAR(255) NOT NULL,
      content TEXT NOT NULL,
      meeting_time TIMESTAMP WITH TIME ZONE,
      location VARCHAR(255),
      created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
      updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP

  CREATE EXTENSION IF NOT EXISTS vector;

      id BIGSERIAL PRIMARY KEY,
      meeting_id BIGINT REFERENCES meeting_minutes(id) ON DELETE CASCADE,
      chunk_number INT,
      chunk_content TEXT NOT NULL,
      embedding vector(1536),
      created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
  );

  - [ ] 创建 FunASR 服务代码
    ```python
    # funasr_service.py
    from fastapi import FastAPI
    from pydantic import BaseModel
    from modelscope.pipelines import pipeline
    from modelscope.utils.constant import Tasks
    
    app = FastAPI()
    
    # 初始化模型
    asr_pipeline = pipeline(
        task=Tasks.auto_speech_recognition,
        model='damo/speech_paraformer-large_asr_nat-zh-cn-16k-common-vocab8404-pytorch'
    )
    
    speaker_model = pipeline(
        task=Tasks.speaker_diarization,
        model='damo/speech_campplus_sv_zh-cn_16k-task-vocab8404-pytorch'
    )
    
    class AudioRequest(BaseModel):
        audio_path: str
        enable_speaker_diarization: bool = True
    
    @app.get("/health")
    async def health_check():
        return {"status": "healthy"}
    
    @app.post("/transcribe")
    async def transcribe_audio(request: AudioRequest):
        result = asr_pipeline(audio_path=request.audio_path)
        
        if request.enable_speaker_diarization:
            # 进行说话人识别
            speaker_result = speaker_model(audio_path=request.audio_path)
            # 合并说话人信息
            combined_result = merge_speaker_info(result, speaker_result)
            return combined_result
        else:
            return result
    ```

## 注意事项

1. **模型下载**：FunASR模型较大，下载时注意网络稳定性
2. **GPU支持**：如果使用GPU，需要配置CUDA环境（可选）
3. **模型资源**：模型可能占用较大内存，确保服务器有足够资源
4. **API Key 安全**：妥善保管 API Key，建议使用环境变量
5. **数据库备份**：定期备份数据库数据
6. **资源限制**：注意文件大小限制，避免服务器资源耗尽
